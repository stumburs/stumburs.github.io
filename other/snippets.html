    • Map

    Useful for mapping from one range of numbers to another e.g. you can map a sin value from -1; 1 to 0; 100. 

    double Map(double n, double start1, double stop1, double start2, double stop2)
    {
        return ((n - start1) / (stop1 - start1)) * (stop2 - start2) + start2;
    }

    Bitshift Multiplication

    Returns a*2n. 

    int BitshiftMultiplication(int a, int n)
    {
        return a <<= n - 1;
    }

    Insertion operator << overloading for a class

    Allows to simply std::cout << SomeClass << std::endl; with specific values.

    If placed OUTSIDE of class 

    std::ostream &operator<<(std::ostream &os, SomeClass &class_object)
    {
        return os << class_object.value;
    }

    If placed INSIDE of class 

    friend std::ostream &operator<<(std::ostream &os, const SomeClass &class_object)
    {
        return os << class_object.value;
    }

    Insertion operator << overloading for a 1D vector

    Allows to simply std::cout << vec << std::endl; of most primary data types. 

    template <typename T>
    std::ostream &operator<<(std::ostream &os, const std::vector<T> &vec)
    {
        for (T i : vec)
            os << i << ' ';
        return os;
    }

    RGB to HEX

    Returns a 6 character long HEX number as std::string from 3 int type RGB values ranging 0-255. 

    #include <iomanip>
    #include <sstream>
    std::string RGBToHEX(int r, int g, int b)
    {
        int values[] = {r, g, b};
        std::stringstream sstream;
        for (int &v : values)
        {
            v = v < 0 ? v = 0 : v > 255 ? v = 255 : v;
            sstream << std::uppercase << std::setw(2) << std::setfill('0') << std::hex << v;
        }
        return sstream.str();
    }

    3D Vector Sorting by Amount

    Sorts a 3 dimensional vector by the amount of elements in the innermost vector.
    Note: this code snippet has been run through ChatGPT to format it properly for HTML, so don't be surprised if it looks weird or doesn't work. 

    std::vector<std::vector<std::vector<int>>> sort_3d_vector_by_amount_of_values(const std::vector<std::vector<std::vector<int>>>& vec)
{
    // Flatten the 3D vector to a 2D vector
    std::vector<std::vector<int>> flat_vec;
    for (const auto& vec2d : vec)
        for (const auto& vec1d : vec2d)
            flat_vec.push_back(vec1d);

    // Sort the 2D vector based on the size of the innermost vectors
    auto sort_by_amount = [](const std::vector<int>& a, const std::vector<int>& b)
    {
        return a.size() < b.size();
    };
    std::sort(flat_vec.begin(), flat_vec.end(), sort_by_amount);

    // Reconstruct the 3D vector from the sorted 2D vector
    std::vector<std::vector<std::vector<int>>> sorted_vec3d;
    int flat_vec_index = 0;
    for (const auto& vec2d : vec)
    {
        std::vector<std::vector<int>> sorted_vec2d;
        for (const auto& vec1d : vec2d)
            sorted_vec2d.push_back(flat_vec[flat_vec_index++]);
        sorted_vec3d.push_back(sorted_vec2d);
    }

    return sorted_vec3d;
}

    Scoped timer

    Outputs scope lifetime in µs
    Instantiate object to start timing. Timing stops when object destructed. 

    #include <iostream>
    #include <chrono>

    struct Timer
    {
        std::chrono::time_point<std::chrono::high_resolution_clock> start, end;

        Timer()
        {
            start = std::chrono::high_resolution_clock::now();
        }

        ~Timer()
        {
            end = std::chrono::high_resolution_clock::now();
            auto ms = std::chrono::duration_cast<std::chrono::microseconds>(end - start);
            std::cout << "Timer took " << ms.count() << "us\n";
        }
    };
